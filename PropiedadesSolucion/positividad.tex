		  \item<2->  $|\gamma_1|\leq\gamma_2$ ,
		  \item<2->  $\alpha_1\gamma_2\leq\alpha_2|\gamma_1|,$
			\item<5-6> $-1< \gamma_1 < 0$ and $0<\gamma_2< 1$, 
			\item<5-6> $ \exists p>1$ t.q
						$\beta_i>\frac{1}{2}p(p-1)\sigma_i$
		 \end{enumerate}
	\end{textblock*}
